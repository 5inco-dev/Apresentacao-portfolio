student-analytics.java

Java + SQL Moderno 
public record StudentProgress(Long studentId, double score, int attendance) {}

String query = """
  SELECT student_id, AVG(score) AS avg_score, SUM(attendance) AS attendance_days
  FROM evaluations
  WHERE semester = ?
  GROUP BY student_id
""";

boolean eligibleForMentoring(StudentProgress p) {
  return p.score() < 7.0 || p.attendance() < 75;
}
